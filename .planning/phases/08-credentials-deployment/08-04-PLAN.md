  <action>
Import the 7 webhook handler workflows first. These must be active before production workflows self-chain into them.

```bash
source /c/Users/DELL/Documents/Antigravity/vision-gridai-platform/.env
WORKFLOWS_DIR="/c/Users/DELL/Documents/Antigravity/vision-gridai-platform/workflows"
N8N_API="https://n8n.srv1297445.hstgr.cloud/api/v1"

WAVE1=(
  "WF_WEBHOOK_STATUS"
  "WF_WEBHOOK_PROJECT_CREATE"
  "WF_WEBHOOK_TOPICS_GENERATE"
  "WF_WEBHOOK_TOPICS_ACTION"
  "WF_WEBHOOK_PRODUCTION"
  "WF_WEBHOOK_PUBLISH"
  "WF_WEBHOOK_SETTINGS"
)

declare -A WF_IDS

for WF in "${WAVE1[@]}"; do
  echo "--- Importing $WF ---"
  RESULT=$(curl -s -X POST "$N8N_API/workflows" \
    -H "X-N8N-API-KEY: $N8N_API_KEY" \
    -H "Content-Type: application/json" \
    -d @"$WORKFLOWS_DIR/${WF}.json")

  WF_ID=$(echo "$RESULT" | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('id','ERROR'))" 2>/dev/null)

  if [ "$WF_ID" = "ERROR" ] || [ -z "$WF_ID" ]; then
    echo "  IMPORT FAILED: $(echo $RESULT | python3 -c 'import sys,json; d=json.load(sys.stdin); print(d.get(\"message\",str(d))[:120])' 2>/dev/null)"
    echo "  Continuing to next workflow..."
    continue
  fi

  echo "  Imported ID: $WF_ID"
  WF_IDS[$WF]=$WF_ID

  # Activate
  ACT=$(curl -s -X PATCH "$N8N_API/workflows/${WF_ID}/activate" \
    -H "X-N8N-API-KEY: $N8N_API_KEY")
  ACTIVE=$(echo "$ACT" | python3 -c "import sys,json; print(json.load(sys.stdin).get('active','?'))" 2>/dev/null)
  echo "  Activated: $ACTIVE"
done
```
